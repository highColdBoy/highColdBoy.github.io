package main

import (
    "fmt"
    "os"
)

// 定义结构体类型 point
type point struct {
    x, y int
}

func main() {
    // 创建结构体实例
    p := point{1, 2}
    
    // 1. 基本格式输出（默认格式）
    // 输出: struct1: {1 2}
    fmt.Printf("struct1: %v\n", p)
    
    // 2. 带字段名的结构体输出
    // 输出: struct2: {x:1 y:2}
    fmt.Printf("struct2: %+v\n", p)
    
    // 3. Go语法表示的结构体（包含包名和类型）
    // 输出: struct3: main.point{x:1, y:2}
    fmt.Printf("struct3: %#v\n", p)
    
    // 4. 输出值的类型
    // 输出: type: main.point
    fmt.Printf("type: %T\n", p)
    
    // 5. 布尔值输出
    // 输出: bool: true
    fmt.Printf("bool: %t\n", true)
    
    // 6. 十进制整数输出
    // 输出: int: 123
    fmt.Printf("int: %d\n", 123)
    
    // 7. 二进制输出
    // 输出: bin: 1110 (14的二进制)
    fmt.Printf("bin: %b\n", 14)
    
    // 8. Unicode字符输出
    // 输出: char: ! (ASCII 33的字符)
    fmt.Printf("char: %c\n", 33)
    
    // 9. 十六进制输出（小写字母）
    // 输出: hex: 1c8 (456的十六进制)
    fmt.Printf("hex: %x\n", 456)
    
    // 10. 浮点数默认输出
    // 输出: float1: 78.900000
    fmt.Printf("float1: %f\n", 78.9)
    
    // 11-12. 科学计数法输出（小写e/大写E）
    // 输出: float2: 1.234000e+08
    // 输出: float3: 1.234000E+08
    fmt.Printf("float2: %e\n", 123400000.0)
    fmt.Printf("float3: %E\n", 123400000.0)
    
    // 13. 字符串输出（不转义）
    // 输出: str1: "string"
    fmt.Printf("str1: %s\n", "\"string\"")
    
    // 14. 带引号的字符串输出（安全转义）
    // 输出: str2: "\"string\""
    fmt.Printf("str2: %q\n", "\"string\"")
    
    // 15. 字符串的十六进制表示（每字节两个字符）
    // 输出: str3: 6865782074686973 ("hex this"的十六进制)
    fmt.Printf("str3: %x\n", "hex this")
    
    // 16. 指针地址输出（十六进制表示）
    // 输出: pointer: 0x...（具体地址值）
    fmt.Printf("pointer: %p\n", &p)
    
    // 17-21. 格式化宽度和对齐控制 ------------------------
    
    // 17. 整数右对齐，宽度6位
    // 输出: width1: |    12|   345|
    fmt.Printf("width1: |%6d|%6d|\n", 12, 345)
    
    // 18. 浮点数右对齐，宽度6位，保留2位小数
    // 输出: width2: |  1.20|  3.45|
    fmt.Printf("width2: |%6.2f|%6.2f|\n", 1.2, 3.45)
    
    // 19. 浮点数左对齐，宽度6位，保留2位小数
    // 输出: width3: |1.20  |3.45  |
    fmt.Printf("width3: |%-6.2f|%-6.2f|\n", 1.2, 3.45)
    
    // 20. 字符串右对齐，宽度6位
    // 输出: width4: |   foo|     b|
    fmt.Printf("width4: |%6s|%6s|\n", "foo", "b")
    
    // 21. 字符串左对齐，宽度6位
    // 输出: width5: |foo   |b     |
    fmt.Printf("width5: |%-6s|%-6s|\n", "foo", "b")
    
    // 22. 格式化字符串到变量（不打印）
    s := fmt.Sprintf("sprintf: a %s", "string")
    fmt.Println(s) // 输出: sprintf: a string
    
    // 23. 格式化输出到指定IO（这里是标准错误）
    // 输出到stderr: io: an error
    fmt.Fprintf(os.Stderr, "io: an %s\n", "error")
}
