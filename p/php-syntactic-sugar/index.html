class User
{
    private string $name;
    private string $gender;

    public function __construct(string $name, string $gender)
    {
        $this->name = $name;
        $this->gender = $gender;
    }

    public function getInfo(): string
    {
        return $this->name . ' ' . $this->gender;
    }
}

class Work
{
    private User $user;

    public function __construct(User $user)
    {
        $this->user = $user;
    }

    public function getUser(): User
    {
        return $this->user;
    }
}

$user = new User('zs', 'male');
$work = (new Work($user));

$userInfo = $work?->getUser()?->getInfo();

$arr1 = ['a' => 'apple', 0 => 'banana'];
$arr2 = ['a' => 'apricot', 0 => 'berry', 1 => 'cherry'];

$result = array_merge($arr1,$arr2); 
// [0 => 'banana',1 => 'berry','cherry','a'=>'apricot']

$result2 = $arr1 + $arr2; 
// [0 => 'banana',1 => 'berry'','a'=>'apricot']
// 直接修改原数组
$arr1 += $arr2

需要覆盖同名键且重新索引数字键 → array_merge
需要保留左侧数据，仅补充新键 → + 运算符
